SET SERVEROUTPUT ON


CREATE TRIGGER New_Employee
BEFORE INSERT
ON EMPLOYEE
FOR EACH ROW
ENABLE
DECLARE
	Admin_User VARCHAR2(20);
BEGIN
	SELECT USER INTO Admin_User FROM dual;
	DBMS_OUTPUT.PUT_LINE('YOU JUST ADDED A NEW EMPLOYEE MR. '|| Admin_User);
END;




CREATE OR REPLACE TRIGGER TR_EMPLOYEE
BEFORE INSERT OR DELETE OR UPDATE ON EMPLOYEE
FOR EACH ROW
ENABLE
DECLARE
    ADMIN_USER VARCHAR(20);
    
BEGIN
    SELECT user INTO ADMIN_USER FROM dual;
    IF INSERTING THEN
        DBMS_OUTPUT.PUT_LINE('ONE ROW INSERTED BY ' ||ADMIN_USER);
    ELSIF DELETING THEN
        DBMS_OUTPUT.PUT_LINE('ONE ROW DELETED BY '||ADMIN_USER);
    ELSIF UPDATING THEN
         DBMS_OUTPUT.PUT_LINE('ONE ROW UPDATED BY '||ADMIN_USER);
    END IF;
END;
