SELECT Employee.name as "Employee Name",Designation as "Employee's Designation",Cage_no,animal.English_name as "Animal Name",catagory.class_name as "Animal's Catagory",caging.nick_name as "Nick Name", addition.add_date as "Addition Date",food.ITEM_NAME as "Food Name",food.SCALE_PER_DAY as "Quantity(Kg)" FROM Employee join works_for using(Employee_Id) join animal using(animal_id) join caging using(animal_id) join cage using(cage_no) join INCLUSION using(cage_no) join addition using(addition_no) join fooding using(animal_id) join food using(food_id) join classification using(animal_id) join catagory using (catagory_id);
